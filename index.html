async function chiamaGemini(prompt, keyIndex) { let currentKey = apiKeys[keyIndex]; // Se la chiave nel file non esiste o è la stringa di esempio if (!currentKey || currentKey.includes("INSERISCI")) { currentKey = window.prompt("Le chiavi su GitHub sono disattivate. Incolla qui la tua chiave API di Google AI Studio:"); if (!currentKey) throw new Error("Devi inserire una chiave per continuare."); } const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${currentKey}`; try { const response = await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [{ parts: [{ text: "Sei un cuoco stellato. Crea una ricetta con: " + prompt }] }] }) }); const data = await response.json(); // CONTROLLO DI SICUREZZA PER EVITARE L'ERRORE '0' if (data.candidates && data.candidates[0] && data.candidates[0].content) { return data.candidates[0].content.parts[0].text; } else { console.error("Dettagli errore Google:", data); throw new Error(data.error ? data.error.message : "Errore: l'AI non ha risposto. Controlla la console."); } } catch (err) { // Se hai più chiavi, prova la successiva if (keyIndex < apiKeys.length - 1) return chiamaGemini(prompt, keyIndex + 1); throw err; } }
