async function chiamaGemini(prompt, keyIndex) { if (keyIndex >= apiKeys.length) { throw new Error("Tutte le chiavi API sono esaurite."); } const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${apiKeys[keyIndex]}`; const requestBody = { contents: [{ parts: [{ text: "Sei un cuoco. Crea una ricetta con: " + prompt }] }] }; const response = await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(requestBody) }); const data = await response.json(); // --- CONTROLLO DI SICUREZZA AGGIUNTO --- if (data.candidates && data.candidates[0] && data.candidates[0].content) { return data.candidates[0].content.parts[0].text; } else { // Se c'è un errore di quota (429), passa alla prossima chiave if (response.status === 429) { return chiamaGemini(prompt, keyIndex + 1); } throw new Error("Gemini non ha generato una risposta. Controlla i filtri di sicurezza o la validità della chiave."); } }
