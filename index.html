async function chiamaGemini(prompt, keyIndex) { // Se le chiavi nel file sono bloccate o rimosse, usiamo una chiave di emergenza inserita a mano let currentKey = apiKeys[keyIndex]; if (!currentKey) { currentKey = prompt("Le chiavi nel file non funzionano. Incolla qui una chiave valida:"); if (!currentKey) throw new Error("Nessuna chiave disponibile."); } const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${currentKey}`; try { const response = await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [{ parts: [{ text: "Sei un cuoco. Crea una ricetta con: " + prompt }] }] }) }); const data = await response.json(); // CONTROLLO ROBUSTO if (data.candidates && data.candidates[0] && data.candidates[0].content) { return data.candidates[0].content.parts[0].text; } else { console.error("Errore API:", data); throw new Error(data.error ? data.error.message : "Risposta vuota dall'AI"); } } catch (err) { if (keyIndex < apiKeys.length - 1) return chiamaGemini(prompt, keyIndex + 1); throw err; } }
